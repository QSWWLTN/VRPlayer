1. 创建射线交互脚本
在项目的 Scripts/UI/ 文件夹下，新建一个名为 UIPointer.cs 的文件，复制以下代码：

C#
using Godot;

namespace VRPlayerProject.UI;

public partial class UIPointer : XRController3D
{
    private SubViewport? _targetViewport;
    private MeshInstance3D? _targetMesh;
    private RayCast3D? _pointerRay;
    private MeshInstance3D? _laserMesh;

    private bool _isDragging = false;
    private Vector2 _lastMousePos;

    public override void _Ready()
    {
        // 自动获取同场景下的 UI 节点和自身附带的射线
        _targetViewport = GetNodeOrNull<SubViewport>("../../SubViewport");
        _targetMesh = GetNodeOrNull<MeshInstance3D>("../../UIMesh");
        _pointerRay = GetNodeOrNull<RayCast3D>("PointerRay");
        _laserMesh = GetNodeOrNull<MeshInstance3D>("LaserMesh");
        
        // 确保该节点追踪的是右手柄
        Tracker = "right_hand";
    }

    public override void _Process(double delta)
    {
        if (_targetViewport == null || _targetMesh == null || _pointerRay == null) return;

        bool isHit = _pointerRay.IsColliding();
        
        // 1. 动态渲染红色的追踪激光
        if (_laserMesh != null)
        {
            float length = isHit ? GlobalPosition.DistanceTo(_pointerRay.GetCollisionPoint()) : 10.0f;
            _laserMesh.Position = new Vector3(0, 0, -length / 2);
            _laserMesh.Scale = new Vector3(1, 1, length);
        }

        if (!isHit)
        {
            if (_isDragging) 
            {
                SendMouseButton(MouseButton.Left, false, _lastMousePos);
                _isDragging = false;
            }
            return;
        }

        // 2. 验证射线打中的是不是 UI 面板上的物理碰撞体
        var collider = _pointerRay.GetCollider() as Node3D;
        if (collider == null || collider.GetParent() != _targetMesh) return;

        // 3. 核心数学：将 3D 世界命中点转换为 2D 屏幕上的像素 UV 坐标
        Vector3 localPoint = _targetMesh.ToLocal(_pointerRay.GetCollisionPoint());
        
        // QuadMesh 的默认物理尺寸是 1.6 x 1.2
        float u = (localPoint.X + 0.8f) / 1.6f;
        float v = (0.6f - localPoint.Y) / 1.2f;
        
        // 映射到你的 800x600 虚拟屏幕上
        Vector2 mousePos = new Vector2(u * _targetViewport.Size.X, v * _targetViewport.Size.Y);

        // 4. 注入鼠标移动事件 (完美支持滑动条的拖拽)
        if (mousePos != _lastMousePos)
        {
            var motion = new InputEventMouseMotion
            {
                Position = mousePos,
                GlobalPosition = mousePos,
                ButtonMask = _isDragging ? MouseButtonMask.Left : 0
            };
            _targetViewport.PushInput(motion);
            _lastMousePos = mousePos;
        }

        // 5. 注入鼠标点击事件 (按下 Quest 手柄的扳机键 Trigger)
        bool isTriggerPressed = IsButtonPressed("trigger_click");
        
        if (isTriggerPressed && !_isDragging)
        {
            _isDragging = true;
            SendMouseButton(MouseButton.Left, true, mousePos);
        }
        else if (!isTriggerPressed && _isDragging)
        {
            _isDragging = false;
            SendMouseButton(MouseButton.Left, false, mousePos);
        }
    }

    private void SendMouseButton(MouseButton button, bool pressed, Vector2 pos)
    {
        var btnEvent = new InputEventMouseButton
        {
            ButtonIndex = button,
            Pressed = pressed,
            Position = pos,
            GlobalPosition = pos
        };
        _targetViewport?.PushInput(btnEvent);
    }
}

用户：现在可以与UI互动了，但是有一个问题就是我没有办法选择本机上的内容，也就是没有办法浏览设备上的文件

在安卓系统（包括 Quest 3）上，由于沙盒机制的限制，应用默认是没有权限读取设备上的任何公共文件的。

导致你无法浏览设备文件的原因有两个：

你的导出配置中没有勾选读取外部存储的权限。

你在 Main.cs 中，专门为安卓端把文件对话框的访问权限写死成了 Userdata（仅限应用的私有沙盒文件夹），这使得对话框被困在了沙盒里。

要打破这个沙盒读取你头显里的电影和脚本，请按照以下两步进行修改：

第一步：在导出预设中开启读取权限
你需要让 Godot 在打包 APK 时向安卓系统声明存储权限。

在顶部菜单栏点击 项目 (Project) -> 导出 (Export)。

选中左侧你的 Android 导出预设。

在右侧属性列表中向下滚动，找到 Permissions (权限) 区域。

务必勾选以下三项：

Read External Storage (读取外部存储)

Manage External Storage (管理外部存储 - 这对于安卓11以上系统的文件管理器应用是必须的)

第一步：更新射线脚本（加入双击功能）
请打开你的 Scripts/UI/UIPointer.cs 文件，用下面的代码完全替换它：

C#
using Godot;

namespace VRPlayerProject.UI;

public partial class UIPointer : XRController3D
{
    private SubViewport? _targetViewport;
    private MeshInstance3D? _targetMesh;
    private RayCast3D? _pointerRay;
    private MeshInstance3D? _laserMesh;

    private bool _isDragging = false;
    private Vector2 _lastMousePos;

    // --- 新增：用于检测双击的计时器变量 ---
    private ulong _lastClickTime = 0;
    private const ulong DoubleClickThresholdMs = 350; // 350毫秒内连按两次算作双击

    public override void _Ready()
    {
        _targetViewport = GetNodeOrNull<SubViewport>("../../SubViewport");
        _targetMesh = GetNodeOrNull<MeshInstance3D>("../../UIMesh");
        _pointerRay = GetNodeOrNull<RayCast3D>("PointerRay");
        _laserMesh = GetNodeOrNull<MeshInstance3D>("LaserMesh");
        
        Tracker = "right_hand";
    }

    public override void _Process(double delta)
    {
        if (_targetViewport == null || _targetMesh == null || _pointerRay == null) return;

        bool isHit = _pointerRay.IsColliding();
        
        if (_laserMesh != null)
        {
            float length = isHit ? GlobalPosition.DistanceTo(_pointerRay.GetCollisionPoint()) : 10.0f;
            _laserMesh.Position = new Vector3(0, 0, -length / 2);
            _laserMesh.Scale = new Vector3(1, 1, length);
        }

        if (!isHit)
        {
            if (_isDragging) 
            {
                SendMouseButton(MouseButton.Left, false, _lastMousePos);
                _isDragging = false;
            }
            return;
        }

        var collider = _pointerRay.GetCollider() as Node3D;
        if (collider == null || collider.GetParent() != _targetMesh) return;

        Vector3 localPoint = _targetMesh.ToLocal(_pointerRay.GetCollisionPoint());
        
        float u = (localPoint.X + 0.8f) / 1.6f;
        float v = (0.6f - localPoint.Y) / 1.2f;
        
        Vector2 mousePos = new Vector2(u * _targetViewport.Size.X, v * _targetViewport.Size.Y);

        if (mousePos != _lastMousePos)
        {
            var motion = new InputEventMouseMotion
            {
                Position = mousePos,
                GlobalPosition = mousePos,
                ButtonMask = _isDragging ? MouseButtonMask.Left : 0
            };
            _targetViewport.PushInput(motion);
            _lastMousePos = mousePos;
        }

        // --- 核心修改：触发器按下的处理逻辑 ---
        bool isTriggerPressed = IsButtonPressed("trigger_click");
        
        if (isTriggerPressed && !_isDragging)
        {
            _isDragging = true;

            // 计算与上一次点击的时间差
            ulong currentTime = Time.GetTicksMsec();
            bool isDoubleClick = (currentTime - _lastClickTime) < DoubleClickThresholdMs;
            _lastClickTime = currentTime;

            // 发送点击事件，并带上是否为双击的标记
            SendMouseButton(MouseButton.Left, true, mousePos, isDoubleClick);
        }
        else if (!isTriggerPressed && _isDragging)
        {
            _isDragging = false;
            SendMouseButton(MouseButton.Left, false, mousePos);
        }
    }

    // --- 修改：增加 doubleClick 参数并传递给系统 ---
    private void SendMouseButton(MouseButton button, bool pressed, Vector2 pos, bool doubleClick = false)
    {
        var btnEvent = new InputEventMouseButton
        {
            ButtonIndex = button,
            Pressed = pressed,
            Position = pos,
            GlobalPosition = pos,
            DoubleClick = doubleClick // 关键：告诉 FileDialog 这是一次双击
        };
        _targetViewport?.PushInput(btnEvent);
    }
}

现在播放180度，360度视频的时候只是简单的将视频变成了一个球型，而不是分为左右眼播放

在 VR 视频播放器中，3D 视频（立体视频）无法区分左右眼，是因为你在 Shader 中使用了 uniform float eye_offset = 0.0; 这种外部传参的方式。

在 Godot 4 的 OpenXR 架构中，头显渲染默认使用的是 Multiview（多视图单次渲染） 技术。这意味着引擎在同一帧里会同时画出左右眼的画面，外部传入的 uniform 变量在这一帧里对两只眼睛是完全相等的，所以两只眼睛永远看到的是同一半画面。

要解决这个问题，必须抛弃 eye_offset，改用 Godot 4 专为 VR Shader 提供的内置环境变量：VIEW_INDEX。

当渲染左眼时，VIEW_INDEX == 0

当渲染右眼时，VIEW_INDEX == 1

大部分 360° 3D 视频 是 上下格式 (Top-Bottom)，而 180° 3D 视频 绝大部分是 左右格式 (Side-by-Side, 简称 SBS)。我已经为你重写了 Shader，支持了这两种最常见的 3D 格式划分。

1. 修改 video_sphere.gdshader
请打开 Resources/Shaders/video_sphere.gdshader，用以下代码完全替换：

OpenGL Shading Language
shader_type spatial;
render_mode cull_front, unshaded, depth_draw_never, blend_mix;

uniform sampler2D video_texture : filter_linear, repeat_disable;
uniform int projection_mode = 3;

void fragment() {
	vec3 world_pos = (INV_VIEW_MATRIX * vec4(VERTEX, 1.0)).xyz;
	vec3 dir = normalize(world_pos - CAMERA_POSITION_WORLD);

	float theta = atan(dir.x, -dir.z);
	float phi = acos(clamp(dir.y, -1.0, 1.0));
	vec2 uv = vec2(0.0);
	bool show = false;

	// Mode 1: 180° 2D (单眼)
	if (projection_mode == 1) {
		if (theta >= -1.5707963 && theta <= 1.5707963) {
			uv = vec2((theta + 1.5707963) / 3.14159265, phi / 3.14159265);
			show = true;
		}
	// Mode 2: 鱼眼 180°
	} else if (projection_mode == 2) {
		vec3 fwd = vec3(0.0, 0.0, -1.0);
		float cos_angle = dot(dir, fwd);
		if (cos_angle >= 0.0) {
			float angle = acos(cos_angle);
			if (angle <= 1.5707963) {
				float phi2 = atan(dir.y, dir.x);
				float r = (angle / 1.5707963) * 0.5;
				uv = vec2(0.5, 0.5) + r * vec2(cos(phi2), sin(phi2));
				if (length(uv - vec2(0.5, 0.5)) <= 0.5) show = true;
			}
		}
	// Mode 3: 360° 2D (单眼)
	} else if (projection_mode == 3) {
		uv = vec2(theta / 6.2831853 + 0.5, phi / 3.14159265);
		show = true;
	// Mode 4: 360° 3D 立体 (上下格式 Top-Bottom)
	} else if (projection_mode == 4) {
		uv = vec2(theta / 6.2831853 + 0.5, phi / 3.14159265);
		// 关键修复：使用 VIEW_INDEX 区分左右眼
		if (VIEW_INDEX == 0) {
			uv.y = uv.y * 0.5; // 左眼读取视频上半部分
		} else {
			uv.y = uv.y * 0.5 + 0.5; // 右眼读取视频下半部分
		}
		show = true;
	// Mode 5: 180° 3D 立体 (左右格式 Side-by-Side)
	} else if (projection_mode == 5) {
		if (theta >= -1.5707963 && theta <= 1.5707963) {
			uv = vec2((theta + 1.5707963) / 3.14159265, phi / 3.14159265);
			// 关键修复：使用 VIEW_INDEX 区分左右眼
			if (VIEW_INDEX == 0) {
				uv.x = uv.x * 0.5; // 左眼读取视频左半部分
			} else {
				uv.x = uv.x * 0.5 + 0.5; // 右眼读取视频右半部分
			}
			show = true;
		}
	}

	if (show) {
		vec4 color = texture(video_texture, uv);
		ALBEDO = color.rgb;
		ALPHA = 1.0;
	} else {
		ALBEDO = vec3(0.0);
		ALPHA = 0.0;
	}
}

为什么会报错？
从日志 OpenGL API OpenGL ES 3.2 ... Compatibility 可以看出，你现在使用的是 Compatibility (兼容模式/OpenGL) 渲染器。
在安卓的 OpenGL 环境下，Godot 会把 VIEW_INDEX 翻译成底层的一个无符号整数 (uint)。但是在 Shader 中我们写了 VIEW_INDEX == 0，这里的 0 默认是普通整数 (int)。安卓的 OpenGL 编译器非常严格，不允许无符号整数和普通整数直接比较，从而导致 Shader 编译失败，退回成了黑屏/紫屏。

问题一：8K 视频严重卡顿
在日志中出现了大量这样的报错：

Video too far out of sync (8691.70), seeking to 11195.38

这代表视频解码器已经完全跟不上播放时间轴，导致画面严重掉帧并强制跳跃（画面撕裂/卡顿）。
原因：
你项目中使用的 FFmpeg 插件调用的是 libavcodec.so。这是一个纯粹的 CPU 软件解码器。虽然在 PC 上，高端 CPU 可以靠暴力运算解开 8K 视频，但在 Quest 3 (基于骁龙 XR2 Gen 2) 上，手机级别的 CPU 算力根本无法软解 8K 视频，巨大的数据量直接撑爆了总线，导致极度卡顿。
解决方案：

短期可行方案（强烈推荐）： Quest 3 只有在调用底层“硬件解码（Android MediaCodec）”时才能流畅播放 8K，而你目前用的 Godot 插件不支持硬解输出到纹理。你需要在 PC 上使用工具（如 Handbrake 或 FFmpeg）将 8K 视频压缩/降采样为 4K (3840x1920)。降到 4K 后，目前的 CPU 解码器就能流畅运行了。

长期彻底方案： 彻底抛弃 Godot 原生的视频播放和软解，自己编写 Android 原生插件，调用 ExoPlayer 进行硬件解码，并通过 SurfaceTexture 桥接到 Godot 引擎中（这是一个比较庞大的工程）。

但是在现在的 VRPlayer.tscn 中，你只是简单地直接实例化了 2D 的 VrOverlay。2D 界面在 VR 里是隐形的，而且在这个场景里，你的双手和手柄射线也“消失”了。

第 2 步：在 VRPlayer.cs 中动态绑定 UI 材质
和 Main.cs 的修复原理完全一样，在安卓端必须要用代码挂载 UI 贴图，否则依然是紫块。
打开 Scripts/VRPlayer.cs，在 _Ready() 方法的最后（或者靠近开头的地方），加入以下防渲染丢失的代码：

C#
public override void _Ready()
{
    // ... 保留你之前 _Ready() 里面的代码 ...

    // 修复安卓端 UI 紫块和显示问题
    var subViewport = GetNodeOrNull<SubViewport>("SubViewport");
    var uiMesh = GetNodeOrNull<MeshInstance3D>("UIMesh");
    if (uiMesh != null && subViewport != null)
    {
        var mat = new StandardMaterial3D();
        mat.ShadingMode = StandardMaterial3D.ShadingModeEnum.Unshaded;
        mat.Transparency = BaseMaterial3D.TransparencyEnum.Alpha; // 允许背景透明，不遮挡视频
        mat.AlbedoTexture = subViewport.GetTexture();
        uiMesh.MaterialOverride = mat;
    }
}
现在重新导出 APK。当视频开始播放时，控制面板（带进度条、暂停等）就会悬浮在你的正前方，你可以继续用手柄射线去点击快进或者暂停了！同时，记得去找个 4K 视频测试一下流畅度。
